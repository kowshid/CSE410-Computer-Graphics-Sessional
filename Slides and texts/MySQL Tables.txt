
CREATE TABLE  patient  (
   patient_id   INT,
   name   	VARCHAR(255),
   mobile   	VARCHAR(255),
   address   	VARCHAR(255),
   dob   		DATE,
   gender   	VARCHAR(1),
   PRIMARY KEY (patient_id)
);

CREATE TABLE  hospital  (
   hospital_id  INT,
   name   	VARCHAR(255),
   address   	VARCHAR(255),
   PRIMARY KEY (hospital_id)
);

CREATE TABLE department	{
   dept_id 	INT,
   name 	VARCHAR(255),
   PRIMARY KEY (dept_id)
);

CREATE TABLE  doctor  (
   doc_id  	INT,
   name  	VARCHAR(255),
   mobile  	VARCHAR(11),
   address  	VARCHAR(255),
   dept_id   	INT,
   speciality   VARCHAR(255),
   hospital_id	INT,
   PRIMARY KEY 	(doc_id),
   FOREIGN KEY 	fk_hos(hospital_id)
   REFERENCES  	hospital(hospital_id),
   FOREIGN KEY 	fk_dept(dept_id)
   REFERENCES 	department(dept_id)
);

CREATE TABLE  appointment  (
   appt_id   	INT,
   patient_id   INT,
   doc_id   	INT,
   time   	DATETIME,
   PRIMARY KEY (appt_id),
   FOREIGN KEY fk_pat(patient_id)
   REFERENCES  patient(patient_id),
   FOREIGN KEY fk_doc(doc_id)
   REFERENCES  doctor(doc_id)
);

CREATE TABLE  prescription   (
   appt_id   	INT,
   serial_no   	INT,
   name   	VARCHAR(255),
   dosage   	VARCHAR(255),
   PRIMARY KEY (appt_id,serial_no),
   FOREIGN KEY fk_appt(appt_id)
   REFERENCES  appointment(appt_id)
);
